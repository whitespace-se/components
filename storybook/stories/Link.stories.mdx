export const Template = ({ text, ...args }) => <Link {...args}>{text}</Link>;

export const CheckboxTemplate = ({ text, ...args }) => (
  <>
    <input type="checkbox" className={visuallyHidden} id="checkbox" />
    <Link {...args} htmlFor="checkbox">
      {text}
    </Link>
  </>
);

    <input
      type="radio"

      id="radio1"
      name="radios"
      value="1"
    />
    <Link {...args} htmlFor="radio1">
      {text}
    </Link> <input
      type="radio"
      className={visuallyHidden}
      id="radio2"
      name="radios"
      value="2"
    />
    <Link {...args} htmlFor="radio2">
      {text}
    </Link>
  </>
);
